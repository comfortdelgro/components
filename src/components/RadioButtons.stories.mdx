# Radio Buttons

Radio buttons are used to select options within an application.

## Usage

Radio buttons are used to select options within an application.

<Preview withToolbar>
  <Story 
    name="Radios"
    argTypes={{
      type: {
        options: Object.values(RADIO_BUTTON_TYPES),
        control: { type: 'radio' },
      },
      disabled: {
        control: { type: 'boolean' },
      },
      labelText: {
        control: { type: 'text' },
      },
      size: {
        control: { type: 'number' },
        defaultValue: 24,
      },
      value: {
        control: { disable: true }
      },
    }}
    args={{
      type: RADIO_BUTTON_TYPES.PRIMARY,
      labelText: "This is a radio button",
      size: 24,
      disabled: false,
    }}>
    {({ type, labelText, size, disabled }) => {
      const [selectedRadioValue, setSelectedRadioValue] = useState("");
      const radioChangeHandler = (value) => {
        setSelectedRadioValue(value);
        console.debug('Radioed', value);
      }
      return  (
        <RadioGroup
          onClickRadioButton={radioChangeHandler}
          selectedValue={selectedRadioValue}>
          <Radio style={{ margin: "16px" }} type={type} value="0" labelText={labelText} size={`${size}px`} disabled={disabled}/>
          <Radio style={{ margin: "16px" }} type={type} value="1" labelText={labelText} size={`${size}px`} disabled={disabled}/>
        </RadioGroup>
      )
    }}
  </Story>
</Preview>
