import { Meta, Story, Preview } from "@storybook/addon-docs";
import { action } from "@storybook/addon-actions";
import { Button } from "./Buttons";

<Meta title="Components/Buttons" component={Button} />

export const argsMap = {
  'Small': 'small',
  'Large': 'large',
  'Primary': 'primary',
  'Secondary': 'secondary',
  'Orange': 'orange',
  'Light': 'light',
  'Light Secondary': 'light-secondary',
};

export const formatModifiers = (size, variant) => {
  return [ argsMap[size], argsMap[variant] ];
}

export const Template = ({ size, variant, ...rest }) => {
  const modifiers = formatModifiers(size, variant);
  // console.debug('modifiers', { modifiers, size, variant, ...rest});
  return <Button modifiers={modifiers} {...rest}
          onClick={action('button-click')}>This is a button</Button>;
}

# Buttons

Buttons are used to trigger actions within an application.

## Usage

Buttons are used to trigger internal actions within your web applications.
