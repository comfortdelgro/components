import {action} from '@storybook/addon-actions'
import {Button, ButtonKind, ButtonSize} from './Button'

<Meta title='Components/Buttons' component={Button} />

export const Template = (props) => {
  return (
    <Button {...props} onClick={action('Clicked!')}>
      {props.value}
    </Button>
  )
}

# Buttons

Buttons are used to trigger actions within an application.

## Usage

Buttons are used to trigger internal actions within your web applications.
