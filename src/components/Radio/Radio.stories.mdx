import {Meta, Story, Preview, Canvas} from '@storybook/addon-docs'
import {action} from '@storybook/addon-actions'
import {RadioGroup, Radio, RadioKind} from './Radio'
import {useState} from 'react'

<Meta title='Form Controls' component={Radio} />

# Radio Buttons

Radio buttons are used to select options within an application.

## Usage

Radio buttons are used to select options within an application.

<Preview withToolbar>
  <Story
    name='Radios'
    argTypes={{
      type: {
        options: [RadioKind.Primary],
        control: {type: 'radio'},
      },
      disabled: {
        control: {type: 'boolean'},
      },
      labelText: {
        control: {type: 'text'},
      },
      size: {
        control: {type: 'number'},
        defaultValue: 24,
      },
      value: {
        control: {disable: true},
      },
    }}
    args={{
      type: RadioKind.Primary,
      labelText: 'This is a radio button',
      size: 24,
      disabled: false,
    }}
  >
    {({type, labelText, size, disabled}) => {
      const [selectedRadioValue, setSelectedRadioValue] = useState('')
      const radioChangeHandler = (value) => {
        action('Changed!')
        setSelectedRadioValue(value)
      }
      return (
        <RadioGroup
          onClickRadioButton={radioChangeHandler}
          selectedValue={selectedRadioValue}
        >
          <Radio
            style={{margin: '16px'}}
            type={type}
            value='0'
            labelText={labelText}
            size={`${size}px`}
            disabled={disabled}
          />
          <Radio
            style={{margin: '16px'}}
            type={type}
            value='1'
            labelText={labelText}
            size={`${size}px`}
            disabled={disabled}
          />
        </RadioGroup>
      )
    }}
  </Story>
</Preview>
